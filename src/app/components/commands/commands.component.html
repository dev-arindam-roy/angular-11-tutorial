Useful Commands:

Setup:

- First Install NODE

- node -v

- npm -v

- Second Install Angular CLI

Commands: (Angular) https://angular.io/cli

npm install -g @angular/cli

ng --version

ng new my-app

ng serve

ng serve -o

ng server

ng serve --port PORT_NUMBER

ng build --prod

ng g c COMPONENET_NAME

ng g c COMPONENET_NAME -s --skipTests

ng g class CLASS_NAME --skipTests

ng g s SERVICE_NAME

ng g m MODULE_NAME

ng g p PIPE_NAME

ng g d DIRECTIVE_NAME

Commands: (TypeScript)

npm install -g typescript

tsc -v

tsc main

tsc main.js --watch

tsc main --watch

node main

Commands: (SASS)

npm install -g sass

sass --version

sass --watch sass/style.scss:css/style.css
